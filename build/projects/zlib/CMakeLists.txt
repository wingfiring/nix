j_project(zlib zlib)

if (WIN32)
	j_aux_add_source(source_files)
	#include_directories(${j_project_root})
	set(CMAKE_C_FLAGS "${CMAKE_C_FLAGS} /W0 /WX-")
	add_library(zlib STATIC ${source_files})
	j_set_target_default(zlib)

	set_property(TARGET zlib PROPERTY FOLDER Library)
else()
	find_package(ZLIB)
	add_library(zlib SHARED IMPORTED GLOBAL)
	set_property(TARGET zlib PROPERTY IMPORTED_LOCATION ${ZLIB_LIBRARIES})  
endif()
